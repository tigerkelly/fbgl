
// Style is an ascii string with no whitespace(s) before or after '='.

Box styles.
	fg = Text foreground color.
	bg = Text background color.
	filled = true/false, true=box filled with fg.

Line styles.
	fg = Text foreground color.
	bg = Text background color.

Label styles.
	fn = Font name, ie "FreeSerif.ttf"
	fs = Font size, ie 96
	fg = Text foreground color.
	bg = Text background color.

Text styles.
	fn= Font name, ie "FreeSerif.ttf"
	fs = Font size, ie 96
	fg = Text foreground color.
	bg = Text background color.

TextField styles.
	fn= Font name, ie "FreeSerif.ttf"
	fs = Font size, ie 96
	fg = Text foreground color.
	bg = Text background color.
	bWidth = borderwidth in pixels
	prompt = Text prompt
	readOnly = true/false

Circle styles.
	fg = Text foreground color.
	bg = Text background color.
	filled = true/false, true=box filled with fg.

Image styles.
	bc = Border color.
	bw = Broder width.

Arc styles.
	fg = Text foreground color.
	bg = Text background color.
	sa = start angle.
	ea = end angle.		// must be larger than the start angle.
	lw = line width.
